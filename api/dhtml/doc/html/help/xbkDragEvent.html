		function init() {
			layer = new xbkLayer();
			with(layer) {
				setLeft( 300 );
				setTop( 100 );
				setWidth( 100 );
				setHeight( 100 );
				setBgColor("silver");
				setBorderLeft( 1,"red");
				setBorderTop( 1,"red");
				setBorderBottom( 1,"black");
				setBorderRight( 1,"black");
				setVis( "show" );
				setInnerHTML( "<font color='red'>drag me!!</color>" );
			}
			drag = new xbkDragEvent();
			drag.enableDragEvent(layer);
		}
		function myclick() {
			//alert("click..");
		}

	<body onload="init()">
