		function init() {
			xbkIconMenu = new xbkIconBrowser( "xbkIconMenu" );

				setHeight( 250 );
				setLeft( 150 );
				setTop( 50 );

				setBorderLeft( 1,"#D6D3CE");
				setBorderRight( 1,"black");
				setBorderBottom( 1,"black");

				sort();
				selectIconById("icon_0");
				
			}
		}
		var xbkIconMenu_COUNT = 10;
		function addIcon() {
			xbkIconMenu.addItem("icon_"+xbkIconMenu_COUNT, ++xbkIconMenu_COUNT+"_b.doc", '../images/xbkIconView/docL.gif',userDefinedFunction );
			xbkIconMenu.sort();
		}
		
		function removeIcon() {
			if (xbkIconMenu.selected) {
				xbkIconMenu.removeItem( xbkIconMenu.selected.id );
			}
		}
