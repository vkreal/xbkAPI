	<script>
		drag = 0	
		move = 0
		var outer = null;
		var inner = null;
		function init() {
			outer = new xbkLayer();
			with( outer ) {
				setLeft( 200 );
				setTop( 100 );
				setWidth( 200 );
				setHeight( 50 );
				setBgColor("silver");
				setVis( "show" );
			}

			inner = new xbkLayer(outer);
			with( inner ) {
				setLeft( outer.getWidth()-25 );
				setTop(25);
				setWidth( 25 );
				setHeight( 25 );
				setBgColor("red");
				setVis( "show" );
			}
			
			drag = new xbkDragEvent();
			drag.enableDragEvent(inner);
			drag.setBoundedToParent(true);

		}
	</script>
	<body onload="init()">
	<h1>
		Example  --> xbkLayer
	</h1>
