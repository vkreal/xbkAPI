	<script>
		var xbkBar = null;
		function init() {
			xbkBar = new xbkProgressBar("xbkBar");
			with ( xbkBar ) {
				setStyleClass("startStyleClass","endStyleClass");
				setLeft( 100 );
				setTop( 100 );
				setWidth( 200 );
				setHeight( 20 );
				setBgColor("red", "silver");
				setBorderLeft( 1,"black");
				setBorderTop( 1,"black");
				setBorderRight( 1,"black");
				setBorderBottom( 1,"black");
				setVis("show");
			}
			startProgressBar(0);
		}

		function startProgressBar(perc) {
			xbkBar.setPercentComplete( perc );
		
			if ( perc <= 100 ) {
				perc++;
				this.setTimeout( "startProgressBar("+perc+")" , 100 );
			}
		}


	</script>
