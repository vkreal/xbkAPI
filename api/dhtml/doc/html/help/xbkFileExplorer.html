	<script>
		var myExplorer = null;
		function init() {
			myExplorer = new xbkFileExplorer("myExplorer", null);
			with ( myExplorer ) {
				setLeft( 150 );
				setTop( 50 );
				setWidth( 400 );
				setHeight( 200 );
				setVis("show");
				setBgColor("#D6D3CE");
				setBorderLeft( 1,"black");
				setBorderTop( 1,"black"); // #D6D3CE
				setBorderRight( 1,"black");
				setBorderBottom( 1,"black");
				
				root_0 = getParentNode("PARENT_0");
				root_1 = getParentNode("PARENT_1");		

				item_0 = getChildNode("CHILD 0");
				item_1 = getChildNode("CHILD 1");
				item_2 = getChildNode("CHILD 2");
				item_3 = getChildNode("CHILD 3");
			
				with(root_0) {
					addChildNode(item_0);
					addChildNode(item_1);
					addChildNode(root_1);		
				}
				with(root_1) {
					addChildNode(item_2);
					addChildNode(item_3);	
				}
				addRoot(root_0);	

			}
		}
		function NEW_HANDLER() {
			//alert("click File");
		}
		
		var CHILD_COUNT = 4;
		function addItem(){
			var node = myExplorer.tree.getSelectedNode();
			if(node) {
				if (node.bFolder) {
					var item = myExplorer.getChildNode("CHILD " + CHILD_COUNT++);
					node.addChildNode(item);
					myExplorer.tree.draw();
					myExplorer.drawIcon(node);
				}	
			}
		}

	</script>
