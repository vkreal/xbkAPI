<script>
	var tree = null;
	function init() {
		tree = new xbkTree( "tree", null );
		var root_0 = null;
		var root_1 = null;	

		var item_0 = null;
		var item_1 = null;
		var item_2 = null;
		var item_3 = null;

		with( tree ) {
			setLeft( 150 );
			setTop( 50 );
			setWidth( 250 );
			setHeight( 200 );
			setBgColor("silver");
			setVis( "show" );

			root_0 = getParentNode("PARENT_0", xbkTREE_NODE_HANDLER);
			root_1 = getParentNode("PARENT_1", xbkTREE_NODE_HANDLER);		

			item_0 = getChildNode("CHILD 0",xbkTREE_NODE_HANDLER);
			item_1 = getChildNode("CHILD 1",xbkTREE_NODE_HANDLER);
			item_2 = getChildNode("CHILD 2",xbkTREE_NODE_HANDLER);
			item_3 = getChildNode("CHILD 3",xbkTREE_NODE_HANDLER);
			
			with(root_0) {
				addChildNode(item_0);
				addChildNode(item_1);
				addChildNode(root_1);		
			}
			with(root_1) {
				addChildNode(item_2);
				addChildNode(item_3);	
			}
			addRoot(root_0);	
		}
	}
	
	var CHILD_COUNT = 4;	

	function xbkTREE_NODE_HANDLER(obj) {
		//alert(obj.text);
	}
	function remove() {
		tree.removeNode( tree.selected.id );
	}
	function addItem(){
		var node = tree.getSelectedNode();
		if(node) {
			if (node.bFolder) {
				var item = tree.getChildNode("NEW CHILD " + CHILD_COUNT++);
				node.addChildNode(item);
				tree.draw();
			}	
		}
	}
</script>
	
<body onload="init()">
